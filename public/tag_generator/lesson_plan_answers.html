export function assetPath(path: string): string {
  const base = import.meta.env.BASE_URL.replace(/\/$/, '');
  const clean = path.startsWith('/') ? path.slice(1) : path;
  return `${base}/${clean}`;
}

  const heroes: Record<string, Hero> = {};
  for (const hero of Object.values(heroFiles)) {
    heroes[hero.normalized_name] = hero;

  const items: Record<string, Item> = {};
  for (const item of Object.values(itemFiles)) {
    items[item.normalized_name] = item;
  }

  return {
    tags:   tagsData as Tag[],
    heroes,
    items,
  };

// src/store/dataStore.ts
import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import type { AppData, DataMode, Tag, Hero, Item } from '../types';
import { loadDefaultData } from '../data/loader';

// Load defaults once at module level — not inside a function.
const defaultData = loadDefaultData();

interface DataState {
  mode:        DataMode;
  customData:  AppData | null;
  getData:          () => AppData;
  setMode:          (mode: DataMode) => void;
  activateCustomMode: () => void;
  resetToDefaults:  () => void;
  updateTags:       (tags: Tag[]) => void;
  updateHero:       (hero: Hero) => void;
  updateItem:       (item: Item) => void;
  exportData:       () => void;
  importData:       (data: AppData) => void;
}

export const useDataStore = create<DataState>()(
  persist(
    (set, get) => ({
      mode:       'defaults',
      customData: null,

      getData: () => {
        const { mode, customData } = get();
        return mode === 'custom' && customData ? customData : defaultData;
      },

      setMode: (mode) => set({ mode }),

      activateCustomMode: () => {
        const { customData } = get();
        if (!customData) {
          // Deep-clone defaults as the starting point for custom data
          set({ mode: 'custom', customData: JSON.parse(JSON.stringify(defaultData)) });
        } else {
          set({ mode: 'custom' });
        }
      },

      resetToDefaults: () => set({ mode: 'defaults', customData: null }),

      updateTags: (tags) => {
        const { customData } = get();
        if (!customData) return;
        set({ customData: { ...customData, tags } });
      },

      updateHero: (hero) => {
        const { customData } = get();
        if (!customData) return;
        set({
          customData: {
            ...customData,
            heroes: { ...customData.heroes, [hero.normalized_name]: hero },
          },
        });
      },

      updateItem: (item) => {
        const { customData } = get();
        if (!customData) return;
        set({
          customData: {
            ...customData,
            items: { ...customData.items, [item.normalized_name]: item },
          },
        });
      },

      exportData: () => {
        const data = get().customData;
        if (!data) return;
        const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });
        const url  = URL.createObjectURL(blob);
        const a   = document.createElement('a');
        a.href     = url;
        a.download = `deadlock-advisor-custom-${new Date().toISOString().slice(0, 10)}.json`;
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
        URL.revokeObjectURL(url);
      },

      importData: (data) => set({ customData: data, mode: 'custom' }),
    }),
    {
      name: 'deadlock-advisor-data',
      // Only persist mode and customData — defaults are always re-loaded from bundled files
      partialize: (state) => ({ mode: state.mode, customData: state.customData }),
    }
  )
);

// src/utils/calculator.ts
import type { Hero, Item, Tag, HeroBuild, TagWeights } from '../types';

export const TIER_MULTIPLIERS: Record<number, number> = {
  800:  1.0,
  1600: 1.2,
  3200: 1.4,
  6400: 1.6,
  9999: 2.0,
};

export interface TeamSetup {
  self:    Hero;
  allies:  Hero[];
  enemies: Hero[];
}

export interface CalculatorConfig {
  allyMult:  number;
  enemyMult: number;
}

export interface ItemScore {
  item:       Item;
  allyScore:  number;
  selfScore:  number;
  enemyScore: number;
  total:      number;
}

export interface BuildResult {
  hero:        Hero;
  build:       HeroBuild;
  buildScore:  number;
  itemScores:  ItemScore[];
}

const DEFAULT_CONFIG: CalculatorConfig = { allyMult: 1.0, enemyMult: 1.0 };

// Σ(weights[tag] × scores[tag]) — the core dot product
function dotProduct(weights: TagWeights, scores: TagWeights): number {
  let sum = 0;
  for (const [tag, weight] of Object.entries(weights)) {
    if (weight == null) continue;
    sum += weight * (scores[tag] ?? 0);
  }
  return sum;
}

export function scoreItem(
  item:   Item,
  build:  HeroBuild,
  config: CalculatorConfig = DEFAULT_CONFIG
): ItemScore {
  const s    = item.values.self_score;
  const v    = build.values;
  const tier = TIER_MULTIPLIERS[item.tier] ?? 1.0;

  const allyScore  = dotProduct(v.ally_weight,  s);
  const selfScore  = dotProduct(v.self_weight,  s);
  const enemyScore = dotProduct(v.enemy_weight, s);
  const total      = (allyScore * config.allyMult + selfScore + enemyScore * config.enemyMult) * tier;

  return { item, allyScore, selfScore, enemyScore, total };
}

export function scoreBuild(
  build:   HeroBuild,
  allies:  Hero[],
  enemies: Hero[]
): number {
  let score = 0;
  for (const ally of allies) {
    for (const ab of ally.builds) {
      score += dotProduct(build.values.ally_weight, ab.values.self_score);
    }
  }
  for (const enemy of enemies) {
    for (const eb of enemy.builds) {
      score += dotProduct(build.values.enemy_weight, eb.values.self_score);
    }
  }
  return score;
}

export function runCalculation(
  setup:  TeamSetup,
  items:  Item[],
  _tags:  Tag[],
  config: CalculatorConfig = DEFAULT_CONFIG
): BuildResult[] {
  return setup.self.builds
    .map(build => {
      const buildScore = scoreBuild(build, setup.allies, setup.enemies);
      const itemScores = items
        .map(item => scoreItem(item, build, config))
        .sort((a, b) => b.total - a.total);
      return { hero: setup.self, build, buildScore, itemScores };
    })
    .sort((a, b) => b.buildScore - a.buildScore);
}

// src/utils/buildPath.ts — PRE-WRITTEN, do not modify
import type { Item } from '../types';
import { TIER_MULTIPLIERS } from './calculator';
import type { ItemScore } from './calculator';

export interface BuildChange {
  action:      'buy' | 'upgrade' | 'sell';
  key:         string;
  components?: string[];
  cost?:       number;
  refund?:     number;
}

export interface PhaseResult {
  phase:          string;
  changes:        BuildChange[];
  assistChanges:  BuildChange[];
  counterChanges: BuildChange[];
}

const BUILD_PHASES = [
  { name: 'Lane',       addBudget: 3200,    totalSlots: 9,  minSlots: 3,  maxSells: 0 },
  { name: 'Early',      addBudget: 6400,    totalSlots: 9,  minSlots: 7,  maxSells: 0 },
  { name: 'Mid',        addBudget: 12800,   totalSlots: 10, minSlots: 9,  maxSells: 2 },
  { name: 'Late',       addBudget: 19800,   totalSlots: 12, minSlots: 11, maxSells: 3 },
  { name: 'Extra Late', addBudget: 1000000, totalSlots: 12, minSlots: 12, maxSells: 5 },
];

const PHASE_TIER_MULTS: Record<string, [number, number, number, number]> = {
  'Lane':       [1.4,  0.8,  0.15, 0.0 ],
  'Early':      [0.85, 0.9,  0.5,  0.05],
  'Mid':        [0.5,  0.75, 1.0,  0.6 ],
  'Late':       [0.1,  0.25, 1.0,  1.5 ],
  'Extra Late': [0.0,  0.1,  0.5,  2.0 ],
};

function getPhaseTierMult(phase: string, tier: number): number {
  const m = PHASE_TIER_MULTS[phase] ?? PHASE_TIER_MULTS['Mid'];
  if (tier <= 800)  return m[0];
  if (tier <= 1600) return m[1];
  if (tier <= 3200) return m[2];
  return m[3];
}

export function computeBuildPath(
  scoredItems: ItemScore[],
  allItems:    Item[]
): PhaseResult[] {
  const itemMap: Record<string, Item> = {};
  allItems.forEach(it => { itemMap[it.normalized_name] = it; });

  const scoredMap: Record<string, ItemScore> = {};
  scoredItems.forEach(s => { scoredMap[s.item.normalized_name] = s; });

  const upgradesTo: Record<string, string[]> = {};
  Object.keys(itemMap).forEach(k => {
    (itemMap[k].upgrades_from || []).forEach(comp => {
      if (!upgradesTo[comp]) upgradesTo[comp] = [];
      upgradesTo[comp].push(k);
    });
  });

  const isSubsumed = (k: string, owned: Set<string>) =>
    (upgradesTo[k] || []).some(u => owned.has(u));

  const chainCost = (k: string, owned: Set<string>): number => {
    const item = itemMap[k];
    if (!item) return Infinity;
    const disc = (item.upgrades_from || [])
      .filter(c => owned.has(c))
      .reduce((s, c) => s + (itemMap[c]?.tier ?? 0), 0);
    return item.tier - disc;
  };

  const emitChain = (k: string, owned: Set<string>, changes: BuildChange[]): number => {
    const item = itemMap[k];
    if (!item) return 0;
    let spent = 0;
    (item.upgrades_from || []).forEach(c => {
      if (!owned.has(c) && itemMap[c]) {
        owned.add(c); spent += itemMap[c].tier;
        changes.push({ action: 'buy', key: c, cost: itemMap[c].tier });
      }
    });
    const comps    = (item.upgrades_from || []).filter(c => owned.has(c));
    const mainCost = item.tier - comps.reduce((s, c) => s + (itemMap[c]?.tier ?? 0), 0);
    comps.forEach(c => owned.delete(c));
    owned.add(k); spent += mainCost;
    changes.push({ action: comps.length ? 'upgrade' : 'buy', key: k, components: comps, cost: mainCost });
    return spent;
  };

  const bpScore = (s: ItemScore, phase: string): number => {
    const tier = itemMap[s.item.normalized_name]?.tier ?? 800;
    const gm   = TIER_MULTIPLIERS[tier] || 1;
    return ((s.allyScore / gm) * 0.75 + (s.selfScore / gm) + (s.enemyScore / gm) * 0.75)
      * getPhaseTierMult(phase, tier);
  };

  function greedyMain(ownedIn: Set<string>, budget: number, maxSlots: number,
    minSlots: number, phase: string, maxSells: number) {
    const owned = new Set(ownedIn);
    let remaining = budget;
    const changes: BuildChange[] = [], soldInPhase = new Set<string>();
    let sellCount = 0;
    for (let iter = 0; iter < 100; iter++) {
      const slots = owned.size, filling = slots < minSlots;
      let bestKey: string | null = null, bestVal = -Infinity;
      for (const k of Object.keys(scoredMap)) {
        const s = scoredMap[k];
        if (owned.has(k) || isSubsumed(k, owned) || soldInPhase.has(k)) continue;
        const cost = chainCost(k, owned);
        if (cost <= 0 || cost > remaining || slots + 1 > maxSlots) continue;
        const ps = bpScore(s, phase);
        if (ps <= 0) continue;
        const val = filling ? ps / cost : ps;
        if (val > bestVal) { bestVal = val; bestKey = k; }
      }
      if (bestKey) { remaining -= emitChain(bestKey, owned, changes); continue; }
      if (maxSells > 0 && sellCount < maxSells && owned.size >= maxSlots) {
        let worstKey: string | null = null, worstPS = Infinity;
        owned.forEach(k => {
          if (soldInPhase.has(k)) return;
          const ps = scoredMap[k] ? bpScore(scoredMap[k], phase) : 0;
          if (ps < worstPS) { worstPS = ps; worstKey = k; }
        });
        if (!worstKey) break;
        const refund = Math.floor((itemMap[worstKey]?.tier ?? 0) / 2);
        const test = new Set(owned); test.delete(worstKey);
        const tb = remaining + refund;
        let swapKey: string | null = null, swapPS = -Infinity;
        for (const k of Object.keys(scoredMap)) {
          const s = scoredMap[k];
          if (test.has(k) || isSubsumed(k, test) || soldInPhase.has(k)) continue;
          const cost = chainCost(k, test);
          if (cost <= 0 || cost > tb || test.size + 1 > maxSlots) continue;
          const ps = bpScore(s, phase);
          if (ps > swapPS) { swapPS = ps; swapKey = k; }
        }
        if (swapKey && swapPS > worstPS * 2.0) {
          owned.delete(worstKey); soldInPhase.add(worstKey);
          remaining += refund; sellCount++;
          changes.push({ action: 'sell', key: worstKey, refund });
        } else { break; }
      } else { break; }
    }
    return { changes, owned, remaining };
  }

  function greedyAssist(mainOwned: Set<string>, globalUsed: Set<string>,
    budget: number, maxSlots: number, key: 'ally' | 'enemy') {
    const tag   = key === 'ally' ? 'assist_importance' : 'counter_importance';
    const local = new Set<string>();
    let remaining = budget;
    const changes: BuildChange[] = [];
    for (let i = 0; i < 10; i++) {
      if (local.size >= maxSlots) break;
      let bestKey: string | null = null, bestScore = -Infinity;
      for (const k of Object.keys(scoredMap)) {
        const s = scoredMap[k];
        if (mainOwned.has(k) || globalUsed.has(k) || local.has(k)) continue;
        if (!itemMap[k] || itemMap[k].tier > remaining) continue;
        if ((s.item.values.self_score[tag] ?? 0) <= 0) continue;
        const score = key === 'ally' ? s.allyScore * 0.75 : s.enemyScore * 0.75;
        if (score > bestScore && score > 0) { bestScore = score; bestKey = k; }
      }
      if (!bestKey) break;
      local.add(bestKey); globalUsed.add(bestKey);
      remaining -= itemMap[bestKey].tier;
      changes.push({ action: 'buy', key: bestKey, cost: itemMap[bestKey].tier });
    }
    return { changes };
  }

  let owned = new Set<string>(), remainingBudget = 0;
  const phaseResults: PhaseResult[] = [];
  const globalAssist = new Set<string>(), globalCounter = new Set<string>();

  for (const phase of BUILD_PHASES) {
    remainingBudget += phase.addBudget;
    const { changes, owned: newOwned, remaining: newBudget } =
      greedyMain(owned, remainingBudget, phase.totalSlots, phase.minSlots, phase.name, phase.maxSells);
    owned = newOwned; remainingBudget = newBudget;
    const side = Math.floor(phase.addBudget / 2);
    phaseResults.push({
      phase: phase.name, changes,
      assistChanges:  greedyAssist(owned, globalAssist,  side, 2, 'ally').changes,
      counterChanges: greedyAssist(owned, globalCounter, side, 2, 'enemy').changes,
    });
  }
  return phaseResults;
}

// src/pages/SettingsPage.tsx
import { useRef } from 'react';
import { useDataStore } from '../store/dataStore';
import { Switch }  from '@/components/ui/switch';
import { Button }  from '@/components/ui/button';
import { Label }   from '@/components/ui/label';
import type { AppData } from '../types';

export default function SettingsPage() {
  const { mode, activateCustomMode, setMode, exportData, importData, resetToDefaults } = useDataStore();
  const fileRef = useRef<HTMLInputElement>(null);

  function handleImport(e: React.ChangeEvent<HTMLInputElement>) {
    const file = e.target.files?.[0];
    if (!file) return;
    const reader = new FileReader();
    reader.onload = (ev) => {
      try {
        const data = JSON.parse(ev.target?.result as string) as AppData;
        importData(data);
        alert('Data imported successfully!');
      } catch {
        alert('Failed to import: invalid JSON file.');
      }
    };
    reader.readAsText(file);
    e.target.value = '';
  }

  function handleReset() {
    if (confirm('This will delete all your custom data and revert to defaults. Are you sure?')) {
      resetToDefaults();
    }
  }

  return (
    <div className="max-w-lg space-y-8">
      <h1 className="text-2xl font-bold">Settings</h1>

      <section className="bg-slate-800 rounded-lg p-6 space-y-4">
        <h2 className="font-semibold text-lg">Data Mode</h2>
        <div className="flex items-center gap-3">
          <Switch
            checked={mode === 'custom'}
            onCheckedChange={(v) => v ? activateCustomMode() : setMode('defaults')}
          />
          <Label>{mode === 'custom' ? 'Using My Custom Data' : 'Using Defaults'}</Label>
        </div>
        <p className="text-slate-400 text-sm">
          In Defaults mode the data is read-only and always reflects the latest deployed weights.
          Switch to Custom to edit heroes, items, and tags — your changes are saved locally in your browser.
        </p>
      </section>

      <section className="bg-slate-800 rounded-lg p-6 space-y-4">
        <h2 className="font-semibold text-lg">Export / Import</h2>
        <div className="flex gap-3">
          <Button onClick={exportData} disabled={mode !== 'custom'} variant="outline">
            Export My Data
          </Button>
          <Button onClick={() => fileRef.current?.click()} variant="outline">
            Import Data
          </Button>
          <input ref={fileRef} type="file" accept=".json" className="hidden"
            onChange={handleImport} />
        </div>
        <p className="text-slate-400 text-sm">
          Export your custom data as a JSON file to back it up or transfer to another device.
          Import a previously exported file to restore your data.
        </p>
      </section>

      <section className="bg-slate-800 rounded-lg p-6 space-y-4">
        <h2 className="font-semibold text-lg text-red-400">Reset</h2>
        <Button variant="destructive" onClick={handleReset}>Reset to Defaults</Button>
        <p className="text-slate-400 text-sm">
          Deletes all custom data from your browser and reverts to the bundled defaults. This cannot be undone — export first if you want to keep your data.
        </p>
      </section>
    </div>
  );
}
